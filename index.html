ViewController.swift
import UIKit

class ViewController: UIViewController {
  override func viewDidLoad() {
    super.viewDidLoad()
    // تهيئة الواجهة
    setupUI()
  }

  private func setupUI() {
    let label = UILabel()
    label.text = "مرحبًا بالعالم!"
    label.textAlignment = .center
    label.frame = view.bounds
    view.addSubview(label)
  }
}
