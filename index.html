سنريك ما لم تره عينك ✨

نعمل جاهدين لإطلاق تجربة iOS مبهرة جداً قريبًا.

SneakPeekViewController.swift
import UIKit

class SneakPeekViewController: UIViewController {
    override func viewDidLoad() {
        super.viewDidLoad()
        setupSneakPeek()
    }

    private func setupSneakPeek() {
        let label = UILabel()
        label.text = "سنريك ما لم تره عينك"
        label.textAlignment = .center
        label.frame = view.bounds
        view.addSubview(label)
    }
}
